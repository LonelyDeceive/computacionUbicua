#include <LiquidCrystal_I2C.h>
LiquidCrystal_I2C lcd(0x27,16,2);

int SENSOR;
float TEMPERATURA;


void setup() {
  Serial.begin(9600);
  lcd.init();
  lcd.backlight();
  lcd.clear();
}

void loop() { 
  SENSOR = analogRead(A0);
  TEMPERATURA = ((SENSOR * 5000.0) / 1023) / 10;
  
  Serial.print("Temperatura: ");
  Serial.println(TEMPERATURA,1);

  lcd.setCursor(0,0);
  lcd.print("Temp: ");
  lcd.print(TEMPERATURA, 1);
  lcd.print("C");

  delay(1000);
}
